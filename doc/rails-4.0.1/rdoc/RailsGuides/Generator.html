def register_kindle_mime_types
  Mime::Type.register_alias("application/xml", :opf, %(opf))
  Mime::Type.register_alias("application/xml", :ncx, %(ncx))

def set_flags_from_environment
  @edge     = ENV['EDGE']     == '1'
  @warnings = ENV['WARNINGS'] == '1'
  @all      = ENV['ALL']      == '1'
  @kindle   = ENV['KINDLE']   == '1'
  @version  = ENV['RAILS_VERSION'] || 'local'
  @lang     = ENV['GUIDES_LANGUAGE']
end
